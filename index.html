Print the count of all alphabet(using map)

    let s = "abcadeecfb";
    const m = new Map();
    for (let i = 0; i < s.length; i++) {
        let char = s.charAt(i);
        if (m.has(char)) {
            let value = m.get(char);
            m.set(char, value + 1);
        }
        else {
            m.set(char, 1);
        }
    }
    for (const [key, value] of m) {
        console.log(key + "=" + value);
    }
